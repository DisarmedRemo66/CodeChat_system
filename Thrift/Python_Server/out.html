Gives appropriate char grade according to the score of the students

void grades(){
    for (int i = 0; i < 20; i++)
    {
        if (studentdata[i].testScore > 89 && studentdata[i].testScore <= 100)
        {
            studentdata[i].grade = 'A';
        }
        else if (studentdata[i].testScore > 79 && studentdata[i].testScore <= 89)
        {
            studentdata[i].grade = 'B';
        }
        else if (studentdata[i].testScore > 69 && studentdata[i].testScore <= 79)
        {
            studentdata[i].grade = 'C';
        }
        else if (studentdata[i].testScore > 59 && studentdata[i].testScore <= 79)
        {
            studentdata[i].grade = 'D';
        }
        else if (studentdata[i].testScore < 60)
        {
            studentdata[i].grade = 'F';
        }
    }
}

Checks for highest score among 20 students and return the highest score

int highest_score(){
    int highest=0;
    for (int k=0; k<20; k++){
        if (studentdata[k].testScore>=highest){
            highest=studentdata[k].testScore;
            }
    }
    return highest;
}

Prints highest test scoring student name and score

 
void highest_test_score_student(int grade){
    cout<<"Highest test scoring students are: " <<endl;

        if (studentdata[i].testScore==grade){
            cout <<studentdata[i].studentLname<<" , "<<studentdata[i].studentFName << " " <<studentdata[i].testScore <<endl;
        }
    }
}
